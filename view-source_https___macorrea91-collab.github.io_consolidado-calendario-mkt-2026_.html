// Fetch or load local fallback
function loadAllData() {
if (cloudApiUrl) {
showToast("Sincronizando con Google Sheets...", "info");
fetch(cloudApiUrl)
.then(response => response.json())
.then(data => {
if (data && !data.error) {
events = data;
updateDashboard();
showToast("Sincronización en la nube completa", "success");
} else {
throw new Error(data.error || "Datos inválidos");
}
})
.catch(error => {
console.error("Error cargando desde Google Sheets:", error);
showToast("Error de conexión. Usando respaldo local.", "error");
loadLocalBackup();
});
} else {
loadLocalBackup();
}
}

function syncWithCloud(action, payload) {
if (!cloudApiUrl) return;
showToast("Actualizando planilla...", "info");
fetch(cloudApiUrl, {
method: "POST",
mode: "no-cors", // Crucial for direct bypass of Google Apps Script redirects
headers: {
"Content-Type": "application/json"
},
body: JSON.stringify({
action: action,
item: payload.item,
id: payload.id
})
})
.then(() => {
showToast("Sincronizado en la nube", "success");
// Refresh data
setTimeout(loadAllData, 1000);
})
.catch(err => {
console.error("Cloud write failed", err);
showToast("Cambio guardado localmente (Nube offline)", "error");
});
}

// Filter engine
function getFilteredEvents() {
return events.filter(e => {
// Search term match
const matchSearch = activeFilters.search === "" ||
e.title.toLowerCase().includes(activeFilters.search) ||
(e.description && e.description.toLowerCase().includes(activeFilters.search)) ||
(e.location && e.location.toLowerCase().includes(activeFilters.search)) ||
(e.media && e.media.toLowerCase().includes(activeFilters.search)) ||
(e.host && e.host.toLowerCase().includes(activeFilters.search));
// Category match
const matchCategory = activeFilters.categories.includes(e.category);
// Segment match
const matchSegment = activeFilters.segments.includes(e.segment);

return matchSearch && matchCategory && matchSegment;
});
}

// CALENDAR BUILDER (Semana inicia el Lunes - Adaptación Dinámica de Altura)
function renderCalendar() {
const calendarGrid = document.getElementById("calendar-grid");
calendarGrid.innerHTML = "";

const year = currentDate.getFullYear();
const month = currentDate.getMonth();

// Set Title text
const monthNames = ["Enero", "Febrero", "Marzo", "Abril", "Mayo", "Junio", "Julio", "Agosto", "Septiembre", "Octubre", "Noviembre", "Diciembre"];
document.getElementById("calendar-title").innerText = `${monthNames[month]} ${year}`;

// Adjust first day for Monday start
let firstDayIndex = new Date(year, month, 1).getDay();
firstDayIndex = (firstDayIndex === 0) ? 6 : firstDayIndex - 1; // Maps Sunday from index 0 to 6, and Monday to 0.

const lastDayDate = new Date(year, month + 1, 0).getDate();
const prevLastDayDate = new Date(year, month, 0).getDate();

const filteredEvents = getFilteredEvents();

// Pre-calculate days and map events to dates
const eventsByDate = {};
filteredEvents.forEach(ev => {
eventsByDate[ev.date] = eventsByDate[ev.date] || [];
eventsByDate[ev.date].push(ev);
});

// 1. Previous month trailing days
for (let i = firstDayIndex; i > 0; i--) {
const dayNum = prevLastDayDate - i + 1;
const formattedMonth = String(month === 0 ? 12 : month).padStart(2, '0');
const formattedYear = month === 0 ? year - 1 : year;
const fullDateStr = `${formattedYear}-${formattedMonth}-${String(dayNum).padStart(2, '0')}`;
calendarGrid.appendChild(createCalendarDayNode(dayNum, true, fullDateStr, []));
}

// 2. Current Month days
for (let i = 1; i <= lastDayDate; i++) {
const fullDateStr = `${year}-${String(month + 1).padStart(2, '0')}-${String(i).padStart(2, '0')}`;
const dayEvents = eventsByDate[fullDateStr] || [];
calendarGrid.appendChild(createCalendarDayNode(i, false, fullDateStr, dayEvents));
}

// 3. Next month trailing days to complete 42 boxes grid structure
const totalCells = firstDayIndex + lastDayDate;
const remainingCells = totalCells % 7 === 0 ? 0 : 7 - (totalCells % 7);
for (let i = 1; i <= remainingCells; i++) {
const formattedMonth = String(month === 11 ? 1 : month + 2).padStart(2, '0');
const formattedYear = month === 11 ? year + 1 : year;
const fullDateStr = `${formattedYear}-${formattedMonth}-${String(i).padStart(2, '0')}`;

calendarGrid.appendChild(createCalendarDayNode(i, true, fullDateStr, []));
}
}

function formatSpanishDate(dateStr) {
const months = ["Ene", "Feb", "Mar", "Abr", "May", "Jun", "Jul", "Ago", "Sep", "Oct", "Nov", "Dic"];
const parts = dateStr.split("-");
if (parts.length === 3) {
const day = parseInt(parts[2]);
const monthIdx = parseInt(parts[1]) - 1;
const year = parts[0];
return `${day} ${months[monthIdx]}, ${year}`;
}
return dateStr;
}
